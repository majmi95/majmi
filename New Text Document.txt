<script>
 const yesBtn = document.getElementById('yesBtn');
 const noBtn  = document.getElementById('noBtn');
 const arena  = document.getElementById('arena');
 const card   = document.getElementById('card');
 const toast  = document.getElementById('toast');
 let tries = 0;
 function showToast(msg){
   toast.textContent = msg;
   toast.classList.add('show');
   clearTimeout(showToast._t);
   showToast._t = setTimeout(()=>toast.classList.remove('show'), 1200);
 }
 function rand(min, max){ return Math.random() * (max - min) + min; }
 function moveNo(){
   tries++;
   // كبّر نعم شوي مع كل محاولة
   const scale = Math.min(1 + tries * 0.12, 2.2);
   yesBtn.style.transform = `scale(${scale})`;
   // رسائل لطيفة
   const msgs = [
     "لا؟ متأكد؟ 😅",
     "لا ما تمشي هنا 😂",
     "حاول مرة ثانية…",
     "وش هالصرامة! 🙈",
     "خلاص اضغط نعم 😍"
   ];
   showToast(msgs[Math.min(tries-1, msgs.length-1)]);
   // لو كثرت المحاولات، نخليه يستسلم
   if(tries >= 8){
     noBtn.textContent = "خلاص 😅";
     noBtn.style.opacity = "0.65";
   }
   // حدود الحركة داخل arena
   const a = arena.getBoundingClientRect();
   const b = noBtn.getBoundingClientRect();
   const padding = 8;
   const maxX = a.width  - b.width  - padding;
   const maxY = a.height - b.height - padding;
   const x = rand(padding, Math.max(padding, maxX));
   const y = rand(padding, Math.max(padding, maxY));
   noBtn.style.left = x + "px";
   noBtn.style.top  = y + "px";
   // اهتزاز خفيف
   card.classList.remove('shake');
   void card.offsetWidth;
   card.classList.add('shake');
 }
 // كمبيوتر: قرب الماوس
 noBtn.addEventListener('mouseenter', moveNo);
 // جوال: لمس
 noBtn.addEventListener('touchstart', (e)=>{ e.preventDefault(); moveNo(); }, {passive:false});
 // أيضًا لو حاول يضغط
 noBtn.addEventListener('click', (e)=>{ e.preventDefault(); moveNo(); });
 yesBtn.addEventListener('click', ()=>{
   celebrate();
   card.innerHTML = `
<div class="final">
<h2>وافقت! ❤️🥰</h2>
<p class="sub">خلاص… تم الحجز رسمي 😄</p>
<div class="badge">🎉 مبروك لنا • ❤️ forever</div>
</div>
   `;
 });
 // قلوب طايرة
 function spawnHeart(){
   const h = document.createElement('div');
   h.className = 'heart';
   h.textContent = Math.random() > 0.5 ? '❤️' : '💖';
   h.style.left = rand(0, 100) + 'vw';
   h.style.fontSize = rand(16, 44) + 'px';
   h.style.setProperty('--drift', rand(-40, 40) + 'vw');
   h.style.animationDuration = rand(3.2, 6.5) + 's';
   document.body.appendChild(h);
   setTimeout(()=>h.remove(), 7000);
 }
 setInterval(spawnHeart, 260);
 // احتفال كثيف وقت "نعم"
 function celebrate(){
   for(let i=0;i<30;i++){
     setTimeout(spawnHeart, i*40);
   }
 }
</script>
